Las palabras son de 32 bits. Aunque los leds/switches tengan 32, solo usan 16.
Los switches usan las posiciones mas altas, y los leds las bajas.


Este codigo lo pensamos para poder debug fuera del lab, hasrdcodeando
// memory-mapped I/O addresses
# GPIO_SWs   = 0x80001400
# GPIO_LEDs  = 0x80001404
# GPIO_INOUT = 0x80001408

.globl main


main:
  li t0, 0x80001400   # En t0 guardamos la direccion GPIO_SWs
  li t1, 0xFFFF       # En t1 guardamos FFFF
                      # upper half = switches (inputs)  (=0)
                      # lower half = outputs (LEDs)     (=1)
  li t0, 0x80001404   # Hardcodeamos el dato
  sw t1, 0(t0)        # Desplazamos 8 en hex a t0 (llega a GPIO_INOUT), y le metemos t1

repeat:
  lw   t1, 0(t0)      # Lee lo que hay en la direcci√≥n de t0, y lo mete a t1
  srli t1, t1, 16     # shift val to the right by 16 bits
  sw   t1, 4(t0)      # Desplazamos t0 hasta GPIO_LEDs, y enchufamos alli t1
  j    repeat         # repeat loop
